            {
              $(".cube").removeClass("flip-to-top").addClass("flip-to-bottom");
              $(this).data("face", "top").text("Flip: to top");
            } 
            else 
            {
              $(".cube").removeClass("flip-to-bottom").addClass("flip-to-top");
              $(this).data("face", "bottom").text("Flip: to bottom");
            }
